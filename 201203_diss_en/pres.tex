\begin{frame}{Order parameter in the B-phase of $^3$He}
Order parameter:
$$
A_{jk} = \Delta(P,T)\ e^{i\phi}\ R_{jk}({\bf n}, \theta).
$$

\bigskip
Energy of spin-orbit interaction:
$$
F_D = \frac{8 \chi \Omega_B^2}{15\gamma^2}
\left(\cos\theta+\frac14\right)^2.
$$

minimum at $\theta = \arccos(-1/4) \approx 104.5^\circ$.
\end{frame}


\begin{frame}{B-phase texture in the cylindrical cell}
 \image{t}{flare_out}{}
 $\alpha = 60^\circ$, $\varphi = \cos^{-1}(1/\sqrt{5})\approx 63.4^\circ$
\end{frame}


\begin{frame}{Leggett equations}
$$
\left\{
\begin{array}{rcl}
\dot{\bf M} &=&
  \displaystyle
  \gamma\ {\bf M} \times {\bf H} +
  \frac{4}{15}\ \frac{\chi\Omega_B^2}{\gamma}\ \sin\theta\ (1+4\cos\theta)\ {\bf n}\\[2ex]
\dot\theta  &=&
  \displaystyle
  \gamma\ {\bf n}\left(\frac{\bf M}{\chi}-{\bf H}\right)\\[2ex]
\dot{\bf n} &=&
  \displaystyle
  -\frac{\gamma}{2}
  \ {\bf n} \times \left(\frac{\bf M}{\chi}-{\bf H}\right)-\\[2ex]
&&\displaystyle
  -\frac{\gamma}{2}
    \ \ctg\frac{\theta}{2}\left[
    {\bf n}\left({\bf n}\left(
      \frac{\bf M}{\chi}-{\bf H}
    \right)\right)-\left(
      \frac{\bf M}{\chi}-{\bf H}
    \right)
  \right]
\end{array}\right.
$$

\bigskip
Frequencies of small oscillations in linear NMR:
$$
\omega_1 = \omega_0 + \frac{\Omega_B^2}{2\omega_0}\sin^2\varphi, \qquad
\omega_2 = \Omega_B \cos\varphi;
$$

$\omega_0 = \gamma H$~--- Larmor frequency,
$\varphi$~--- angle between ~$\bf n$ and~$\bf H$.
\end{frame}

\begin{frame}{Results}
\begin{itemize}

\item впервые получена формула для низкочастотной моды
пространственно-однородных колебаний ОПД;

\item показано, что измерение частоты этих колебаний может быть
использовано для измерения леггеттовской частоты в B-фазе $^3$He;

\item с помощью численных экспериментов было исследовано влияние
пространственной неоднородности и магнитной релаксации на частоту
и коэффициент затухания этих колебаний;

\item колебания были обнаружены экспериментально как в объемном
$^3$He, так и в $^3$He в аэрогеле, причем их свойства качественно
совпали с теорией. Для объемного $^3$He, параметры которого известны,
наблюдается и хорошее количественное совпадение с теорией.

\item полученные результаты позволили провести измерения Леггеттовской
частоты B-фазы $^3$He в аэрогеле.

\end{itemize}
\end{frame}
